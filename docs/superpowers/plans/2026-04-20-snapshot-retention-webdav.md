# Snapshot Retention + WebDAV Upload — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Cap local snapshot disk usage, collapse rapid-fire snapshots, and opt-in upload each new snapshot to WebDAV using a collision-safe `{slug}-{hash8}/project.json` namespace.

**Architecture:** Extend the existing `GlobalSettings` + project-TOML overlay with a `[snapshot]` section (global + project). Rewrite `create_snapshot` in `core/src/services/snapshots.rs` to a pure-function `create_snapshot_inner(project_dir, name, now_secs, config)` that does: scan existing snapshots → plan replace-vs-append → atomic write via `.pending` dir + rename → prune oldest if over cap → fire-and-log WebDAV upload. WebDAV upload lives in a new `core/src/services/snapshot_webdav.rs` behind `#[cfg(feature = "sync")]`, reusing `services::webdav` primitives and `services::sync::read_sync_config`.

**Tech Stack:** Rust (tokio, walkdir, blake3, reqwest via services::webdav), Svelte 5 runes, tauri-specta-generated IPC bindings, Vitest for unit tests, Playwright for E2E.

**Spec:** `docs/superpowers/specs/2026-04-20-snapshot-retention-webdav-design.md`

pub fn resolve(
    global: &GlobalSettings,
    project_view: Option<&ViewConfig>,
    project_new_file: Option<&NewFileConfig>,
    project_plugins: Option<&PluginsConfig>,
    project_snapshot: Option<&SnapshotConfig>,
) -> EffectiveSettings {
    // ... existing view/new_file/plugins resolution unchanged ...

    let snapshot = ResolvedSnapshot {
        max_count: project_snapshot
            .and_then(|s| s.max_count)
            .or(global.snapshot.max_count)
            .unwrap_or(DEFAULT_SNAPSHOT_MAX_COUNT),
        min_interval_minutes: project_snapshot
            .and_then(|s| s.min_interval_minutes)
            .or(global.snapshot.min_interval_minutes)
            .unwrap_or(DEFAULT_SNAPSHOT_MIN_INTERVAL_MINUTES),
    };

    EffectiveSettings {
        view,
        new_file,
        plugins: ResolvedPlugins { enabled },
        snapshot,
        is_project_scoped: project_view.is_some()
            || project_new_file.is_some()
            || project_plugins.is_some()
            || project_snapshot.is_some(),
    }
}

fn validate_snapshot_config(s: &SnapshotConfig) -> Result<(), AppError> {
    if let Some(n) = s.max_count {
        if n == 0 || n > 10_000 {
            return Err(AppError::InvalidInput(format!(
                "snapshot.max_count must be between 1 and 10000, got {n}"
            )));
        }
    }
    if let Some(m) = s.min_interval_minutes {
        if m > 10_080 {
            return Err(AppError::InvalidInput(format!(
                "snapshot.min_interval_minutes must be ≤ 10080 (one week), got {m}"
            )));
        }
    }
    Ok(())
}

pub async fn create_snapshot_inner(
    project_dir: &str,
    name: &str,
    now_secs: u64,
    _config: ResolvedSnapshot,
) -> Result<SnapshotMeta, AppError> {
    let base = snapshots_dir(project_dir);

    // click), remove it — we're about to replace it atomically anyway.
    if canonical.exists() {

    }

        tokio::fs::remove_dir_all(&pending).await?;
    }
    let files_dir = pending.join("files");
    tokio::fs::create_dir_all(&files_dir).await?;

    let mut file_count: usize = 0;
    let mut total_bytes: u64 = 0;

    for entry in WalkDir::new(project_dir).into_iter().filter_map(|e| e.ok()) {
        let path = entry.path();

            continue;

            .map_err(|e| AppError::Custom(e.to_string()))?;
        if relative

        if let Some(parent) = dest.parent() {

        }
        let content = tokio::fs::read(path).await?;
        total_bytes += content.len() as u64;
        tokio::fs::write(&dest, &content).await?;
        file_count += 1;
    }

    let meta = SnapshotMeta {
        id: snap_id,
        name: name.to_string(),
        timestamp: now_secs,
        file_count,
        total_bytes,
    };
    let meta_path = pending.join("metadata.json");

    tokio::fs::write(&meta_path, meta_json).await?;

async fn cleanup_stale_pending(base: &std::path::Path) -> Result<(), AppError> {
    if !base.exists() {
        return Ok(());
    }
    let mut entries = tokio::fs::read_dir(base).await?;
    while let Some(entry) = entries.next_entry().await? {
        let name = entry.file_name();
        if let Some(s) = name.to_str() {
            if s.ends_with(".pending") {
                let _ = tokio::fs::remove_dir_all(entry.path()).await;
            }
        }
    }
    Ok(())
}

//! WebDAV-side snapshot operations: collision-safe naming, identity check,
//! upload, and delete. All functions are best-effort — failures bubble to
//! the caller as `AppError` so the caller can log and continue.
//!
//! Remote layout:
//!
//! ```text
//! {base_url}/novelist-snapshots/{slug}-{hash8}/
//!     project.json                  — identity card
//!     snap-{ts}/
//!         metadata.json
//!         files/...
//! ```

use crate::services::webdav::{self, WebDavAuth};
use crate::AppError;
use reqwest::Client;
use serde::{Deserialize, Serialize};
use std::path::Path;

pub const REMOTE_ROOT: &str = "novelist-snapshots";

#[derive(Debug, Serialize, Deserialize, PartialEq, Eq)]
pub struct ProjectIdentity {
    pub original_path: String,
    pub display_name: String,
    pub created_iso: String,
}

/// Sanitize a folder basename for use in a URL path segment.
///
/// Keeps CJK and word chars; replaces `/ \ : * ? " < > |` and whitespace
/// with `_`. Returns `"project"` if the result is empty or all underscores.
pub fn sanitize_slug(name: &str) -> String {
    let replaced: String = name
        .chars()
        .map(|c| {
            if c == '/'
                || c == '\\'
                || c == ':'
                || c == '*'
                || c == '?'
                || c == '"'
                || c == '<'
                || c == '>'
                || c == '|'
                || c.is_whitespace()
            {
                '_'
            } else {
                c
            }
        })
        .collect();
    if replaced.is_empty() || replaced.chars().all(|c| c == '_') {
        "project".to_string()
    } else {
        replaced
    }
}

/// First 8 hex chars of blake3(project_dir_abs). Stable across restarts;
/// changes only if the absolute path changes (user moved/renamed the folder).
pub fn project_hash8(project_dir: &str) -> String {
    let hash = blake3::hash(project_dir.as_bytes()).to_hex();
    hash.as_str()[..8].to_string()
}

/// Build the per-project remote base path, relative to the WebDAV base URL.
pub fn project_remote_base(project_dir: &str) -> String {
    let name = Path::new(project_dir)
        .file_name()
        .and_then(|s| s.to_str())
        .unwrap_or("project");
    let slug = sanitize_slug(name);
    let hash = project_hash8(project_dir);
    format!("{}/{}-{}", REMOTE_ROOT, slug, hash)
}

/// Check whether the remote project dir is owned by this project.
///
/// Returns:
/// - `Ok(true)` — dir doesn't exist yet, or exists and `project.json` matches. Safe to proceed.
/// - `Ok(false)` — dir exists but `project.json.original_path` is someone else's. Caller should abort upload.
/// - `Err(..)` — network / auth / parse failure. Caller decides.
pub async fn verify_identity(
    client: &Client,
    base_url: &str,
    project_dir: &str,
    auth: &WebDavAuth,
) -> Result<bool, AppError> {
    let remote_base = project_remote_base(project_dir);
    let identity_path = format!("{}/project.json", remote_base);
    // Try to fetch project.json. Missing → treat as empty slot (caller writes one).
    let url = format!(
        "{}/{}",
        base_url.trim_end_matches('/'),
        identity_path.trim_start_matches('/')
    );
    let resp = client
        .get(&url)
        .basic_auth(&auth.username, Some(&auth.password))
        .send()
        .await
        .map_err(|e| AppError::Custom(format!("WebDAV GET project.json failed: {e}")))?;
    if resp.status().as_u16() == 404 {
        return Ok(true); // empty slot
    }
    if !resp.status().is_success() {
        return Err(AppError::Custom(format!(
            "WebDAV GET {} returned {}",
            url,
            resp.status()
        )));
    }
    let body = resp
        .text()
        .await
        .map_err(|e| AppError::Custom(format!("WebDAV project.json read failed: {e}")))?;
    let identity: ProjectIdentity = match serde_json::from_str(&body) {
        Ok(i) => i,
        Err(_) => {
            // Malformed or pre-spec — act like empty slot; write a fresh one.
            return Ok(true);
        }
    };
    Ok(identity.original_path == project_dir)
}

/// Ensure the project's remote dir + identity card exist. Writes `project.json`
/// if missing. Caller is expected to have verified identity first.
pub async fn ensure_project_dir(
    client: &Client,
    base_url: &str,
    project_dir: &str,
    auth: &WebDavAuth,
) -> Result<(), AppError> {
    let _ = webdav::create_collection(client, base_url, REMOTE_ROOT, auth).await;
    let remote_base = project_remote_base(project_dir);
    let _ = webdav::create_collection(client, base_url, &remote_base, auth).await;

    // Only write project.json if missing.
    let identity_path = format!("{}/project.json", remote_base);
    let url = format!(
        "{}/{}",
        base_url.trim_end_matches('/'),
        identity_path.trim_start_matches('/')
    );
    let head = client
        .get(&url)
        .basic_auth(&auth.username, Some(&auth.password))
        .send()
        .await
        .map_err(|e| AppError::Custom(format!("WebDAV GET identity failed: {e}")))?;
    if head.status().is_success() {
        return Ok(());
    }

    let display_name = Path::new(project_dir)
        .file_name()
        .and_then(|s| s.to_str())
        .unwrap_or("project")
        .to_string();
    let identity = ProjectIdentity {
        original_path: project_dir.to_string(),
        display_name,
        created_iso: iso_now(),
    };
    let body = serde_json::to_string_pretty(&identity)
        .map_err(|e| AppError::Custom(format!("serialize identity: {e}")))?;
    let put = client
        .put(&url)
        .basic_auth(&auth.username, Some(&auth.password))
        .body(body)
        .send()
        .await
        .map_err(|e| AppError::Custom(format!("WebDAV PUT identity failed: {e}")))?;
    if !put.status().is_success() && put.status().as_u16() != 201 && put.status().as_u16() != 204 {
        return Err(AppError::Custom(format!(
            "WebDAV PUT identity returned {}",
            put.status()
        )));
    }
    Ok(())
}

/// Seconds-since-epoch as a plain string. Stored in `project.json` for
/// human debugging only — never parsed. Keeping it a plain integer avoids
/// pulling a date crate and dodges TZ/format bikeshedding.
fn iso_now() -> String {
    let secs = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs();
    format!("epoch-{}", secs)
}

/// Upload the entire local `files_dir/` tree for a snapshot to WebDAV.
/// `metadata.json` is uploaded last so that a partial upload isn't detectable
/// as a complete snapshot by future listers.
pub async fn upload_snapshot(
    client: &Client,
    base_url: &str,
    project_dir: &str,
    snap_id: &str,
    local_snap_dir: &Path,
    auth: &WebDavAuth,
) -> Result<(), AppError> {
    // Identity check first. Bail if someone else owns the remote.
    let owned = verify_identity(client, base_url, project_dir, auth).await?;
    if !owned {
        return Err(AppError::Custom(format!(
            "WebDAV path {} is owned by a different project (project.json mismatch)",
            project_remote_base(project_dir)
        )));
    }
    ensure_project_dir(client, base_url, project_dir, auth).await?;

    let remote_snap = snap_remote_dir(project_dir, snap_id);
    webdav::create_collection(client, base_url, &remote_snap, auth).await?;

    let remote_files_root = format!("{}/files", remote_snap);
    webdav::create_collection(client, base_url, &remote_files_root, auth).await?;

    let local_files = local_snap_dir.join("files");
    if !local_files.exists() {
        return Err(AppError::Custom(format!(
            "Local snapshot files dir missing: {}",
            local_files.display()
        )));
    }

    // Walk local files_dir, MKCOL each parent, PUT each file.
    for entry in walkdir::WalkDir::new(&local_files)
        .into_iter()
        .filter_map(|e| e.ok())
    {
        let path = entry.path();
        if !path.is_file() {
            continue;
        }
        let relative = path.strip_prefix(&local_files).map_err(|e| {
            AppError::Custom(format!("strip_prefix failed in upload_snapshot: {e}"))
        })?;
        let rel_str = relative.to_string_lossy().replace('\\', "/");

        // Ensure every parent dir on remote.
        if let Some(parent) = std::path::Path::new(&rel_str).parent() {
            if !parent.as_os_str().is_empty() {
                let parent_remote = format!("{}/{}", remote_files_root, parent.display());
                let _ = webdav::create_collection(client, base_url, &parent_remote, auth).await;
            }
        }

        let remote_path = format!("{}/{}", remote_files_root, rel_str);
        webdav::upload_file(client, base_url, &remote_path, path, auth).await?;
    }

    // metadata.json last.
    let local_meta = local_snap_dir.join("metadata.json");
    if local_meta.exists() {
        let remote_meta = format!("{}/metadata.json", remote_snap);
        webdav::upload_file(client, base_url, &remote_meta, &local_meta, auth).await?;
    }
    Ok(())
}

/// Best-effort remote deletion of a snapshot dir. Errors are logged, never
/// returned.
pub async fn delete_snapshot_remote(
    client: &Client,
    base_url: &str,
    project_dir: &str,
    snap_id: &str,
    auth: &WebDavAuth,
) {
    let remote_snap = snap_remote_dir(project_dir, snap_id);
    if let Err(e) = webdav::delete_remote(client, base_url, &remote_snap, auth).await {
        tracing::warn!(
            "Failed to delete remote snapshot {}/{}: {}",
            project_dir,
            snap_id,
            e
        );
    }
}
